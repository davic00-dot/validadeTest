// ===================== CAMERA & BARCODE ===================== async function openCamera() { const modal = document.getElementById('cameraModal'); modal.classList.add('open'); try { videoStream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment', width: { ideal: 1280 }, height: { ideal: 720 } } }); document.getElementById('video').srcObject = videoStream; // Iniciar scanning melhorado startAdvancedBarcodeScanning(); } catch (err) { console.error('Erro ao acessar câmera:', err); toast('❌ Não foi possível acessar a câmera', true); closeCamera(); } } function closeCamera() { const modal = document.getElementById('cameraModal'); modal.classList.remove('open'); if (scanInterval) clearInterval(scanInterval); if (videoStream) { videoStream.getTracks().forEach(t => t.stop()); videoStream = null; } } async function startAdvancedBarcodeScanning() { const codeReader = new ZXing.BrowserMultiFormatReader(); const videoElement = document.getElementById('video'); let consecutiveFailures = 0; const maxFailures = 30; let lastScannedCode = null; // Melhorar zoom e foco se disponível try { const track = videoStream.getVideoTracks()[0]; const capabilities = track.getCapabilities(); if (capabilities.zoom) { const zoom = capabilities.zoom.max / 2; await track.applyConstraints({ zoom: zoom }); } if (capabilities.focusMode && capabilities.focusMode.includes('continuous')) { await track.applyConstraints({ focusMode: 'continuous' }); } } catch (e) { console.log('Ajustes de câmera não disponíveis'); } scanInterval = setInterval(async () => { try { const result = await codeReader.decodeFromVideoElement(videoElement); if (result) { const code = result.text.trim(); // Evitar leitura duplicada imediata if (code === lastScannedCode) return; lastScannedCode = code; // Validar código lido const cleanCode = code.replace(/\D/g, ''); if (cleanCode.length === 13 || cleanCode.length === 8) { document.getElementById('codigoEAN').value = code; onEanInput(code); closeCamera(); toast('✓ Código lido: ' + code); consecutiveFailures = 0; } else { // Código inválido - continuar tentando consecutiveFailures++; } } else { consecutiveFailures++; } // Parar após muitas falhas if (consecutiveFailures >= maxFailures) { // Relax e continuar tentando com menos frequência consecutiveFailures = 0; } } catch (err) { consecutiveFailures++; // Silenciar erros de leitura } }, 300); // Mais rápido (300ms vs 500ms) } function digitarManual() { closeCamera(); document.getElementById('codigoEAN').focus(); } function onEanInput(value) { const statusEl = document.getElementById('eanStatus'); const input = document.getElementById('codigoEAN'); if (!value) { statusEl.style.display = 'none'; input.style.borderColor = 'var(--border)'; return; } // Validar EAN-13 e EAN-8 const cleanValue = value.replace(/\D/g, ''); if (cleanValue.length === 13) { // Validar checksum EAN-13 const isValid = validateEAN13(cleanValue); if (isValid) { statusEl.style.display = 'block'; statusEl.style.background = 'rgba(0, 229, 160, 0.1)'; statusEl.style.color = 'var(--accent)'; statusEl.textContent = `✓ EAN-13 válido`; input.style.borderColor = 'var(--accent)'; } else { statusEl.style.display = 'block'; statusEl.style.background = 'rgba(255, 107, 53, 0.1)'; statusEl.style.color = 'var(--warn)'; statusEl.textContent = `⚠ EAN-13 inválido (checksum)`; input.style.borderColor = 'var(--warn)'; } } else if (cleanValue.length === 8) { // Validar checksum EAN-8 const isValid = validateEAN8(cleanValue); if (isValid) { statusEl.style.display = 'block'; statusEl.style.background = 'rgba(0, 229, 160, 0.1)'; statusEl.style.color = 'var(--accent)'; statusEl.textContent = `✓ EAN-8 válido`; input.style.borderColor = 'var(--accent)'; } else { statusEl.style.display = 'block'; statusEl.style.background = 'rgba(255, 107, 53, 0.1)'; statusEl.style.color = 'var(--warn)'; statusEl.textContent = `⚠ EAN-8 inválido (checksum)`; input.style.borderColor = 'var(--warn)'; } } else if (cleanValue.length > 0) { statusEl.style.display = 'block'; statusEl.style.background = 'rgba(255, 107, 53, 0.1)'; statusEl.style.color = 'var(--warn)'; statusEl.textContent = `⚠ ${cleanValue.length} dígitos (esperado: 8 ou 13)`; input.style.borderColor = 'var(--warn)'; } else { statusEl.style.display = 'none'; input.style.borderColor = 'var(--border)'; } } // Validar EAN-13 (algoritmo de checksum) function validateEAN13(ean) { if (ean.length !== 13) return false; let sum = 0; for (let i = 0; i < 12; i++) { const digit = parseInt(ean[i], 10); sum += digit * (i % 2 === 0 ? 1 : 3); } const checksum = (10 - (sum % 10)) % 10; return checksum === parseInt(ean[12], 10); } // Validar EAN-8 (algoritmo de checksum) function validateEAN8(ean) { if (ean.length !== 8) return false; let sum = 0; for (let i = 0; i < 7; i++) { const digit = parseInt(ean[i], 10); sum += digit * (i % 2 === 0 ? 3 : 1); } const checksum = (10 - (sum % 10)) % 10; return checksum === parseInt(ean[7], 10); }
